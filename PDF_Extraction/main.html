def main(pdf_path, output_dir, lattice_type):
    """
    Main function to execute the PDF processing workflow.
    Defines the directory to load PDF files from and initiates their processing.
    Returns:
        None
    """

    if not os.path.isdir(output_dir):
        os.makedirs(output_dir)
        print(f"Directory created at {output_dir}")

    status, message, result = process_pdf(pdf_path, output_dir, lattice_type)

    if status == SuccessCode.S200:
        result = result.replace('\n', ' ', regex=True)
        result.columns = [str(col).replace('\n', ' ') for col in result.columns]

        delimiter = ' |||'
        end = ' |||\n'

        output_lines = result.apply(lambda x: delimiter.join(x.astype(str)), axis=1)
        txt_file = os.path.join(output_dir, f"{os.path.splitext(os.path.basename(pdf_path))[0]}.txt")

        with open(txt_file, 'w') as file:
            file.write(delimiter.join(result.columns.astype(str)) + end)
            for line in output_lines:
                file.write(line + end)

        print(f"{status}: {message}")
    else:
        print(f"{status}: {message}")

    gc.collect()

def process_pdf(pdf_path, output_dir, lattice_type):
    """
    Processes a single PDF file using the configured processing strategy.
    Args:
        pdf_path (str): The path to the PDF file to be processed.
        output_dir (str): The path to the output directory where result is to be stored.
        lattice_type (enum): The category of the lattice for choosing processing class.
    """

    processing_strategy = None
    try:
        if lattice_type == LatticeType.LATTICE:
            processing_strategy = ProcessLatticeData()
        elif lattice_type == LatticeType.NONLATTICE_HLM:
            processing_strategy = ProcessNonLatticeHLMData()
        elif lattice_type == LatticeType.NONLATTICE_VLM:
            processing_strategy = ProcessNonLatticeVLMData()
        elif lattice_type ==LatticeType.ENCRYPTED:
            raise CustomError.PdfIsEncryptedError(pdf_path)
        else:
            raise CustomError.InvalidLatticeTypeError(pdf_path)

    except CustomError.InvalidLatticeTypeError as e:
        status = e.code
        message = e.__str__()
        result = pd.DataFrame()

        return status, message, result

    except CustomError.PdfIsEncryptedError as e:
        status = e.code
        message = e.__str__()
        result = pd.DataFrame()

        return status, message, result

    except Exception as e:
        status = ErrorCode.E500
        message = f"{pdf_path} {ErrorCode.E500.Message} {e}"
        result = pd.DataFrame()

        return status, message, result

    processing_context = ProcessingContext(processing_strategy)
    status, message, result = processing_context.process_data(pdf_path, output_dir)

    del processing_strategy
    del processing_context

    return status, message, result
