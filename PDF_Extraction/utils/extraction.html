class WordExtractor:
    """
    This class is used to extract words from a PDF page.
    """

    def __init__(self, page):
        self.__page = page
        self.__words = None

    def __combine_chars_into_word_dict(self, word_chars):
        """
        Combines characters into a word based on the spacing between them.
        """
        word_text = pdfplumber.utils.text.extract_text(word_chars)
        x0 = word_chars[0]["x0"]
        x1 = word_chars[-1]["x1"]
        top = min(char["top"] for char in word_chars)
        bottom = max(char["bottom"] for char in word_chars)
        bbox = (x0, top, x1, bottom)
        height = bottom - top
        width = x1 - x0
        doctop = top

        word_dict = {
            "text": word_text,
            "x0": x0,
            "x1": x1,
            "top": top,
            "doctop": doctop,
            "bottom": bottom,
            "bbox": bbox,
            "upright": word_chars[0]["upright"],
            "height": height,
            "width": width,
            "direction": "ltr",
        }

        return word_dict

    def __combine_words_into_block_dict(self, block_words, pdf_path):
        """
        Combines words into a block based on the spacing between them.
        """
        sorted_words = sorted(block_words, key=lambda word: word["x0"])
        for i in range(len(sorted_words) - 1):
            word1 = sorted_words[i]
            word2 = sorted_words[i + 1]

            if word1["width"] == 0 or word2["width"] == 0:
                continue
            if word1["height"] == 0 or word2["height"] == 0:
                continue

            overlap_width = max(
                0, min(word1["x1"], word2["x1"]) - max(word1["x0"], word2["x0"])
            )
            overlap_height = max(
                0,
                min(word1["bottom"], word2["bottom"]) - max(word1["top"], word2["top"]),
            )
            if min(word1["width"], word2["width"]) == 0:
                continue
            if (
                overlap_width / min(word1["width"], word2["width"]) >= 0.4
                and overlap_height / min(word1["height"], word2["height"]) >= 0.25
            ):
                raise CustomError.OverlappingWordsError(pdf_path)

        word_text = " ".join([word["text"] for word in block_words])
        x0 = block_words[0]["x0"]
        x1 = block_words[-1]["x1"]
        top = min(word["top"] for word in block_words)
        bottom = max(word["bottom"] for word in block_words)
        bbox = (x0, top, x1, bottom)
        height = bottom - top
        width = x1 - x0
        doctop = top

        block_dict = {
            "text": word_text,
            "x0": x0,
            "x1": x1,
            "top": top,
            "doctop": doctop,
            "bottom": bottom,
            "bbox": bbox,
            "upright": block_words[0]["upright"],
            "height": height,
            "width": width,
            "direction": "ltr",
        }

        return block_dict

    def __group_and_create_word_dicts(self, sorted_chars, spacing_threshold=1):
        """
        Groups characters into words based on the spacing between them.
        """
        words = []
        current_word_chars = []

        for _, char in enumerate(sorted_chars):
            if char["text"].strip() == "":
                continue

            if not current_word_chars:
                current_word_chars.append(char)
            else:
                previous_char = current_word_chars[-1]
                spacing = char["x0"] - previous_char["x1"]

                if spacing <= spacing_threshold and spacing >= -1 * spacing_threshold:
                    current_word_chars.append(char)
                else:
                    words.append(
                        self.__combine_chars_into_word_dict(current_word_chars)
                    )
                    current_word_chars = [char]

        if current_word_chars:
            words.append(self.__combine_chars_into_word_dict(current_word_chars))

        return words

    def extract_words(self):
        """
        Extracts words from the PDF page.
        """
        chars = self.__page.chars

        unique_chars = []
        seen = set()

        for char in chars:
            char_tuple = frozenset(char.items())
            if char_tuple not in seen:
                seen.add(char_tuple)
                unique_chars.append(char)

        lines = {}
        for char in unique_chars:
            lines.setdefault(char["top"], []).append(char)
        lines = dict(sorted(lines.items()))

        combined_lines = {}

        last_line_key = None
        last_line_chars = []

        for key, chars in lines.items():
            heights = [char["height"] for char in chars]
            if heights:
                tolerance = max(1, 0.25 * (sum(heights) / len(heights)))
            else:
                tolerance = 1

            if last_line_key is None or abs(key - last_line_key) > tolerance:
                if last_line_key is not None:
                    combined_lines[last_line_key] = last_line_chars
                last_line_key = key
                last_line_chars = chars

                for char in last_line_chars:
                    char["top"] = last_line_key
            else:
                for char in chars:
                    insert_index = next(
                        (
                            i
                            for i, c in enumerate(last_line_chars)
                            if char["x0"] < c["x0"]
                        ),
                        len(last_line_chars),
                    )
                    last_line_chars.insert(insert_index, char)
                    char["top"] = last_line_key

        if last_line_key is not None:
            combined_lines[last_line_key] = last_line_chars

        words = []
        for key, line in combined_lines.items():
            distances = [
                line[i]["x0"] - line[i - 1]["x1"]
                for i in range(1, len(line))
                if 0 < (line[i]["x0"] - line[i - 1]["x1"]) <= 1
            ]

            if not distances:
                tolerance = 0.85
            else:
                tolerance = 2 * (sum(distances) / len(distances))
                tolerance = min(max(0.6, tolerance), 1)

            words.extend(self.__group_and_create_word_dicts(line, tolerance))

        self.__words = words

    def get_words(self):
        """
        Returns the extracted words from the PDF page.
        """
        if self.__words is None:
            self.extract_words()

            if not self.__words:
                return []

        return self.__words

    def extract_text_from_tables(self, tables, pdf_path):
        """
        Extracts text from tables in the PDF page.
        """
        if self.__words is None:
            self.extract_words()

            if not self.__words:
                return []

        extracted_text_from_tables = []
        if tables:
            for table in tables:
                cell_data = {}
                self.__adjust_words_for_columns(table, pdf_path)
                diff_word_tolerance = 1

                for cell in table.cells:
                    if cell is None:
                        continue

                    cell_words = self.__get_cell_words(
                        cell, diff_word_tolerance, pdf_path
                    )
                    cell_blocks = self.__get_cell_blocks(cell_words, pdf_path)
                    cell_data[(cell[0], cell[1])] = self.__combine_cell_blocks(
                        cell_blocks
                    )

                if cell_data:
                    extracted_text_from_tables.append(
                        self.__create_grid_from_cell_data(cell_data)
                    )

        return extracted_text_from_tables

    def __adjust_words_for_columns(self, table, pdf_path):
        """
        Adjusts words for columns in the table.
        """
        for word in self.__words:
            for col in table.columns:
                if (
                    word["x0"] < col.bbox[0]
                    and (0.28 * col.bbox[0] + 0.72 * col.bbox[2]) < word["x1"]
                    and word["top"] > col.bbox[1]
                    and word["bottom"] < col.bbox[3]
                    and 0 < col.bbox[0] - word["x0"] < 15
                ):
                    word["x0"] += col.bbox[0] - word["x0"]
                    word["x1"] += col.bbox[0] - word["x0"]

                if (
                    word["x0"] < col.bbox[0]
                    and word["x1"] > (0.8 * col.bbox[0] + 0.2 * col.bbox[2])
                    and word["top"] > col.bbox[1]
                    and word["bottom"] < col.bbox[3]
                ):
                    if 0 < col.bbox[0] - word["x0"] < 10:
                        word["x0"] += col.bbox[0] - word["x0"]
                        word["x1"] += col.bbox[0] - word["x0"]
                    else:
                        if (0.3 * col.bbox[0] + 0.7 * col.bbox[2]) < word["x1"]:
                            raise CustomError.TableExtractionError(pdf_path)

    def __get_cell_words(self, cell, diff_word_tolerance, pdf_path):
        """
        Gets words within a cell.
        """
        cell_words = []
        for word in self.__words:
            if (
                word["x0"] > cell[0] - diff_word_tolerance
                and word["x0"] < cell[2] - diff_word_tolerance
                and word["top"] > cell[1] - diff_word_tolerance
                and word["top"] < cell[3] - diff_word_tolerance
            ):
                if word["bottom"] > cell[3] + 0.5 * word["height"]:
                    raise CustomError.TableExtractionError(pdf_path)
                cell_words.append(word)
        return cell_words

    def __get_cell_blocks(self, cell_words, pdf_path):
        """
        Gets blocks of words within a cell.
        """
        cell_blocks = []
        current_block = []
        for word in cell_words:
            if not current_block:
                current_block.append(word)
            else:
                last_word = current_block[-1]
                tolerance = 0.25 * (word["bottom"] - word["top"])
                if (
                    last_word["top"] <= word["top"] <= last_word["bottom"] - tolerance
                    or last_word["top"] + tolerance
                    <= word["bottom"]
                    <= last_word["bottom"]
                    or (
                        last_word["top"] >= word["top"]
                        and last_word["bottom"] <= word["bottom"]
                    )
                ):
                    insert_index = next(
                        (
                            i
                            for i, c in enumerate(current_block)
                            if word["x0"] < c["x0"]
                        ),
                        len(current_block),
                    )
                    current_block.insert(insert_index, word)
                else:
                    cell_blocks.append(
                        self.__combine_words_into_block_dict(current_block, pdf_path)
                    )
                    current_block = [word]
        if current_block:
            cell_blocks.append(
                self.__combine_words_into_block_dict(current_block, pdf_path)
            )
        return cell_blocks

    def __combine_cell_blocks(self, cell_blocks):
        """
        Combines cell blocks into a single string.
        """
        temp = {}
        for word in cell_blocks:
            if word["top"] not in temp:
                temp[word["top"]] = word["text"]
            else:
                temp[word["top"]] += f" {word['text']}"
        temp = dict(sorted(temp.items()))
        return "\n".join(temp.values())

    def __create_grid_from_cell_data(self, cell_data):
        """
        Creates a grid from cell data.
        """
        rows = defaultdict(list)
        for (x, y), text in cell_data.items():
            rows[y].append((x, text))

        all_x_coords = sorted({x for x, y in cell_data.keys()})

        grid = []
        for y in sorted(rows.keys()):
            row = [""] * len(all_x_coords)
            for x, text in rows[y]:
                col_index = all_x_coords.index(x)
                row[col_index] = text
            grid.append(row)

        return grid

    def extract_blocks(self, pdf_path):
        """
        Extracts blocks of words from the PDF page.
        """
        if self.__words is None:
            self.extract_words()

            if not self.__words:
                return []

        lines = defaultdict(list)
        for word in self.__words:
            lines[word["top"]].append(word)

        blocks = []
        for key, val in lines.items():
            line_words = sorted(val, key=lambda w: w["x0"])

            distances = [
                line_words[i]["x0"] - line_words[i - 1]["x1"]
                for i in range(1, len(line_words))
                if 0 < (line_words[i]["x0"] - line_words[i - 1]["x1"]) <= 10
            ]

            if not distances:
                tolerance = 5
            else:
                tolerance = 1.2 * (sum(distances) / len(distances))
                tolerance = min(max(3, tolerance), 10)

            current_block = []
            for word in line_words:
                if not current_block:
                    current_block.append(word)
                else:
                    last_word = current_block[-1]
                    if word["x0"] <= last_word["x1"] + tolerance:
                        current_block.append(word)
                    else:
                        blocks.append(
                            self.__combine_words_into_block_dict(
                                current_block, pdf_path
                            )
                        )
                        current_block = [word]

            if current_block:
                blocks.append(
                    self.__combine_words_into_block_dict(current_block, pdf_path)
                )

        return blocks
