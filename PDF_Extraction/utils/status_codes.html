    E600 = {"Code": 600, "Message": "Invalid Data Error"}
    E601 = {"Code": 601, "Message": "Key Error"}

    E700 = {"Code": 700, "Message": "PDF Parsing Error"}

    E800 = {"Code": 800, "Message": "Processing Error"}
    E801 = {"Code": 801, "Message": "DataFrame Merging Error"}
    E802 = {"Code": 802, "Message": "DataFrame Cleaning Error"}
    E803 = {"Code": 803, "Message": "Column Mismatch Error"}

    E805 = {"Code": 805, "Message": "Invalid grid Error"}
    E806 = {"Code": 806, "Message": "Invalid Lattice Type Error"}
    E807 = {"Code": 807, "Message": "Missing Pages during extraction or merging"}
    E808 = {"Code": 808, "Message": "Rows with multiline description exceeds limit."}
    E809 = {"Code": 809, "Message": "Rows with multiple transactions present."}

    E815 = {
        "Code": 815,
        "Message": "Single cell on last cell (Speicific to Deutsche Bank)",
    }
    E816 = {"Code": 816, "Message": "Junk data in last row."}
    E817 = {"Code": 817, "Message": "Multiple junk dates in date cell."}
    E818 = {"Code": 818, "Message": "Table in the form of an image."}
    E819 = {"Code": 819, "Message": "Rows missing due to incorrect date"}
    E820 = {"Code": 820, "Message": "File is encrypted"}
    E821 = {"Code": 821, "Message": "Serial number is invalid or missing"}
    E822 = {"Code": 822, "Message": "Columns missing in the table"}

    def Code(self):
