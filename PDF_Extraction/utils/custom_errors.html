class CustomError:
    """
    Class containing custom exceptions for the PDF extraction utility.
    """
    class CustomErrorWrapper(Exception):
        """
        Base class for all custom exceptions.
        """
        def __init__(self, message, errorcode=None):
            super().__init__(message)
            self.code = errorcode
    class FileNotFoundError(CustomErrorWrapper):
        """
        Exception raised when the file is not found at the specified location.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The file {pdf_file} was not found at the specified location.",
                ErrorCode.E404,
            )
    class PDFParsingError(CustomErrorWrapper):
        """
        Exception raised when the PDF file could not be parsed properly by pdfplumber.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The file {pdf_file} could not be parsed properly by pdfplumber.",
                ErrorCode.E700,
            )
    class ProcessingError(CustomErrorWrapper):
        """
        Exception raised when the PDF file could not be processed properly.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The file {pdf_file} could not be extracted properly by processing.",
                ErrorCode.E800,
            )
    class DataFrameMergingError(CustomErrorWrapper):
        """
        Exception raised when the dataframes corresponding to the PDF file could not be merged properly.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The dataframes corresponding to file {pdf_file} could not be merged properly.",
                ErrorCode.E801,
            )
    class DataFrameCleaningError(CustomErrorWrapper):
        """
        Exception raised when the dataframes corresponding to the PDF file could not be cleaned properly.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The dataframes corresponding to file {pdf_file} could not be cleaned properly.",
                ErrorCode.E802,
            )
    class ColumnMismatchError(CustomErrorWrapper):
        """
        Exception raised when the dataframes corresponding to the PDF file have different number of columns.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The dataframes corresponding to file {pdf_file} have different number of columns.",
                ErrorCode.E803,
            )
    class MultiRowsUnderSameDate(CustomErrorWrapper):
        """
        Exception raised when the dataframes corresponding to the PDF file have multiple transaction rows under same date.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The pdf file {pdf_file} have multiple transaction rows under same date.",
                ErrorCode.E804,
            )
    class InvalidGridError(CustomErrorWrapper):
        """
        Exception raised when the PDF file does not have a valid grid.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The pdf file {pdf_file} does not have valid grid.", ErrorCode.E805
            )
    class InvalidLatticeTypeError(CustomErrorWrapper):
        """
        Exception raised when the lattice type for the PDF file does not qualify.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The provided lattice type for the file {pdf_file} does not qualify.",
                ErrorCode.E806,
            )
    class MissingPagesError(CustomErrorWrapper):
        """
        Exception raised when the extracted dataframes from the PDF file have some missing pages.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The extracted dataframes from the file {pdf_file} have some missing pages.",
                ErrorCode.E807,
            )
    class MultiLineDescriptionExceedingLimitError(CustomErrorWrapper):
        """
        Exception raised when the number of rows in extracted dataframes from the PDF file with multiline description exceeds limit.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The number of rows in extracted dataframes from the file {pdf_file} with multiline description exceeds limit.",
                ErrorCode.E808,
            )
    class MultipleTransactionsError(CustomErrorWrapper):
        """
        Exception raised when the extracted dataframes from the PDF file have multiple transactions within the same row.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The {pdf_file} contains atleast one row with multiple transactions within the same row.",
                ErrorCode.E809,
            )
    class AlternateRowMissError(CustomErrorWrapper):
        """
        Exception raised when the extracted dataframes from the PDF file have alternate rows missing.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The pdf file {pdf_file} has alternate rows missing.", ErrorCode.E810
            )
    class InvalidTransactionsError(CustomErrorWrapper):
        """
        Exception raised when the extracted dataframes from the PDF file have invalid transactions within the same row.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The {pdf_file} contains atleast one row with invalid transactions within the same row.",
                ErrorCode.E811,
            )
    class TableExtractionError(CustomErrorWrapper):
        """
        Exception raised when the tables from the PDF file could not be extracted properly.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The tables from the {pdf_file} could not be extracted properly.",
                ErrorCode.E812,
            )
    class ClosingBalanceInHeader(CustomErrorWrapper):
        """
        Exception raised when the header in the PDF file contains closing balance.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The header in {pdf_file} contains closing balance.", ErrorCode.E813
            )
    class OverlappingWordsError(CustomErrorWrapper):
        """
        Exception raised when the table cells in the PDF file contain overlapping words.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The {pdf_file} contains some overlapping words in the table cells.",
                ErrorCode.E814,
            )
    class SingleCellOnLastPageError(CustomErrorWrapper):
        """
        Exception raised when the last page in the PDF file contains only one single cell which could not be merged with rest of the data.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The {pdf_file} contains one single cell on last page which could not be merged with rest of the data.",
                ErrorCode.E815,
            )
    class JunkDataInLastRowError(CustomErrorWrapper):
        """
        Exception raised when the last row in the PDF file contains some junk data.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"The {pdf_file} contains some junk data in the last row.",
                ErrorCode.E816,
            )
    class MultipleJunkDatesError(CustomErrorWrapper):
        """
        Exception raised when the date cells in the PDF file contain multiple junk dates.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"Some of the date cells in {pdf_file} contains extra junk dates.",
                ErrorCode.E817,
            )
    class TableAsImageError(CustomErrorWrapper):
        """
        Exception raised when the table in the PDF file is in the form of an image.
        """
        def __init__(self, pdf_file):
            super().__init__(f"the table is an image in {pdf_file}", ErrorCode.E818)
    class RowMismatchError(CustomErrorWrapper):
        """
        Exception raised when some rows in the PDF file are missing due to incorrect date.
        """
        def __init__(self, pdf_file):
            super().__init__(
                f"Some rows in the pdf are missing due to incorrect date {pdf_file}",
                ErrorCode.E819,
            )
    class PdfIsEncryptedError(CustomErrorWrapper):
        """
        Exception raised when the PDF file is encrypted.
        """
        def __init__(self, pdf_file):
            super().__init__(f"{pdf_file} is encrypted.", ErrorCode.E820)
    class InvalidSerialNoError(CustomErrorWrapper):
        """
        Exception raised when the serial number in the PDF file is invalid.
        """
        def __init__(self, pdf_file):
            super().__init__(f"Serial number in {pdf_file} is invalid.", ErrorCode.E821)
    class ColumnBreakError(CustomErrorWrapper):
        """
        Exception raised when the columns in the table are missing.
        """
        def __init__(self, pdf_file):
            super().__init__(f"Serial number in {pdf_file} is invalid.", ErrorCode.E822)

    class InvalidDatesError(CustomErrorWrapper):
        """
        The dates column has junk data.
        """
        def __init__(self, pdf_file):
            super().__init__(f"Dates in {pdf_file} are invalid.", ErrorCode.E823)
